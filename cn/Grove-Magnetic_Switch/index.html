{
    int sensorValue = digitalRead(MAGNECTIC_SWITCH);

    {
        return true;//yes,return ture
    }
    else
    {
        return false;//no,return false
    }
}

{
    digitalWrite(LED,HIGH);
}

{
    digitalWrite(LED,LOW);
